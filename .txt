____________Апроксимація у визначені а і б, підставити сюди свої значення

% Дані
I2_I1 = [2.9773; 1.5289; 1.2715; 1.1643; 1.1058; 1.069; 1.0439; 1.0257; 1.0119; 1.0013];
I3_I1 = [4.5521; 1.9440; 1.4804; 1.2875; 1.1821; 1.1159; 1.0707; 1.0379; 1.0132; 0.99404];

% Лінійна апроксимація: y = a + b*x
coeffs = polyfit(I2_I1, I3_I1, 1);  % Повертає [b, a]
b = coeffs(1);
a = coeffs(2);

% Побудова графіка
x_fit = linspace(min(I2_I1), max(I2_I1), 100);
y_fit = a + b * x_fit;

figure;
plot(I2_I1, I3_I1, 'ob', 'MarkerSize', 6, 'DisplayName', 'Дані');
hold on;
plot(x_fit, y_fit, '-r', 'LineWidth', 1.5, 'DisplayName', sprintf('y = %.4f + %.4f·x', a, b));
title('Лінійна апроксимація: I3/I1 = a + b·(I2/I1)');
xlabel('I2/I1');
ylabel('I3/I1');
legend('Location', 'best');
grid on;

% Виведення коефіцієнтів
fprintf('a = %.4f\n', a);
fprintf('b = %.4f\n', b);
